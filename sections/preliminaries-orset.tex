\subsection{Observed-Remove Set (\ORset{})}	\label{ss:orset}

\begin{align}
  \mathcal{F}_{\orset}(\rd, E, \op, \vis, \ar)
  = \set{a \mid\; &\exists e \in E. \op(e) = \add(a) \\
  \land\; \big(&\forall f \in E. \op(f) = \remove(a) \implies \lnot(\visord{e}{f})\big)}.
\end{align}
